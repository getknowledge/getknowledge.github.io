React Redux 简单使用

安装几个库

npm install –save prop-types
npm install –save react-redux
npm install –save redux

添加相应js文件

  1. Action.js

export const ADD_REIMBURSE =’ADD_REIMBURSE’;//添加
export const DEL_REIMBURSE =’DEL_REIMBURSE’;//删除

export function AddBeimburse(data) {
return{
type:ADD_REIMBURSE,
payload:{data}
}
}

export function DelBeimburse(id) {
return{
type:DEL_REIMBURSE,
payload:{id}
}
}

  1. Reducer.js

import {ADD_REIMBURSE,DEL_REIMBURSE} from ../Action.js;

const initialState={
addbaoxiao:[],
}

export default function (state=initialState,action) {

switch(action.type){
    case ADD_REIMBURSE:{
        var newData = state.addbaoxiao.concat(action.payload.data)
        return{
            ...state,
            addbaoxiao:newData
        }
    }
    case DEL_REIMBURSE:{
        var newData = state.addbaoxiao;
        newData.splice(action.payload.id,1);
        return{
            ...state,
            addbaoxiao:newData
        }
    }
default:
        return state;
}

}

  1. Store.js

import { createStore,applyMiddleware } from “redux”;
import thunk from ‘redux-thunk’;
import rootReducer from ‘./reducers’;

let store = createStore(rootReducer,applyMiddleware(thunk));

export default store;
